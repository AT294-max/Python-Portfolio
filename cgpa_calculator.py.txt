import tkinter as tk

def calculate_cgpa():
    grades = [float(entry.get()) for entry in entries]
    cgpa = sum(grades) / len(grades)
    result_label.config(text=f"CGPA: {cgpa:.2f}")

root = tk.Tk()
root.title("CGPA Calculator")
entries = [tk.Entry(root) for _ in range(5)]
for e in entries: e.pack()
tk.Button(root, text="Calculate", command=calculate_cgpa).pack()
result_label = tk.Label(root)
result_label.pack()
root.mainloop()
